Advanced Enemy AI Behavior

This system uses a finite state machine with perception components to create intelligent enemy behaviors. Enemies can patrol areas, investigate sounds, and engage in combat with varying tactics based on their type and health status. 

// Enemy AI State Machine implementation
void AEnemyCharacter::UpdateAIState()
{
    // Determine current state based on perception and health
    float currentHealth = GetCurrentHealth();
    bool playerDetected = PerceptionComponent->IsPlayerDetected();
    
    if (currentHealth < lowHealthThreshold) {
        // When low on health, switch to defensive behavior
        SetCurrentState(EAIState::Defensive);
        ExecuteDefensiveManeuvers();
    }
    else if (playerDetected) {
        // When player is detected, engage based on distance
        float distanceToPlayer = CalculateDistanceToPlayer();
        
        if (distanceToPlayer < meleeRange) {
            SetCurrentState(EAIState::MeleeAttack);
            ExecuteMeleeAttack();
        }
        else if (distanceToPlayer < rangedAttackRange) {
            SetCurrentState(EAIState::RangedAttack);
            ExecuteRangedAttack();
        }
        else {
            SetCurrentState(EAIState::Pursuit);
            ExecutePursuit();
        }
    }
    else if (CurrentState != EAIState::Patrol) {
        // Return to patrol when player is lost
        SetCurrentState(EAIState::Patrol);
        RestartPatrolPath();
    }
}

Custom 2D Game Engine

A lightweight engine built from scratch using C++ and SDL2, featuring a simple entity-component system, collision detection, and basic rendering capabilities. The demo shows a simple platformer game built with the engine. 

// Entity Component System implementation in custom engine
class Entity
{
private:
    std::unordered_map<std::type_index, std::unique_ptr<Component>> components;
    bool active = true;
    EntityID id;

public:
    Entity(EntityID entityId) : id(entityId) {}
    
    template<typename T, typename... Args>
    T* AddComponent(Args&&... args)
    {
        std::type_index typeId = std::type_index(typeid(T));
        
        // Ensure component doesn't already exist
        if (components.find(typeId) != components.end()) {
            return static_cast<T*>(components[typeId].get());
        }
        
        // Create and store new component
        auto component = std::make_unique<T>(std::forward<Args>(args)...);
        T* componentPtr = component.get();
        
        // Initialize component and add to map
        componentPtr->entity = this;
        componentPtr->Initialize();
        components[typeId] = std::move(component);
        
        return componentPtr;
    }
    
    template<typename T>
    T* GetComponent()
    {
        std::type_index typeId = std::type_index(typeid(T));
        auto it = components.find(typeId);
        
        if (it != components.end()) {
            return static_cast<T*>(it->second.get());
        }
        
        return nullptr;
    }
    
    // Additional entity functionality...
};
